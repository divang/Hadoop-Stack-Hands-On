========== Set the mysql root password in CDH VM =========
sudo /etc/init.d/mysqld stop
sudo mysqld_safe --skip-grant-tables &
mysql -uroot

mysql>use mysql;
mysql>update user set authentication_string=PASSWORD("mynewpassword") where User='root';
mysql>flush privileges;
mysql>quit

sudo /etc/init.d/mysqld start
mysql -u root -p

Store to specific Hadoop directory:
sqoop import \
--connect jdbc:mysql://localhost/testDb \
--username root \
--table student \
--m 1 \
--where "name='Aaditya'" \
--target-dir /student_from_mysql

Incremental import:
mysql> insert into student values(4,'Sudhanshu');
mysql> insert into student values(5,'Nitya');

sqoop import \
--connect jdbc:mysql://localhost/testDb \
--username root \
--table student \
--m 1 \
--incremental append \
--check-column id \
--last-value 3
